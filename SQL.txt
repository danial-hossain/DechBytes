-- display_order কলাম যোগ করুন (ইতিমধ্যে যোগ হয়ে থাকলে error দেবে, সেটা নরমাল)
IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'Banners' AND COLUMN_NAME = 'display_order')
BEGIN
    ALTER TABLE Banners ADD display_order INT DEFAULT 0;
    PRINT '✅ display_order column added';
END
ELSE
BEGIN
    PRINT '✅ display_order column already exists';
END
GO

-- existing banners এর জন্য display_order আপডেট করুন
UPDATE Banners SET display_order = id WHERE display_order IS NULL OR display_order = 0;
GO

-- এখন টেবিলের স্ট্রাকচার চেক করুন

GO

-- সব ডাটা দেখুন (সঠিক সিনট্যাক্স সহ)

FROM Banners 
ORDER BY display_order ASC, id DESC;
